Total Cost MoM = 
VAR ThisMonth = CALCULATE([Total Cost], DATEADD(dim_date[date].[Date], 0, MONTH))
VAR LastMonth = CALCULATE([Total Cost], DATEADD(dim_date[date].[Date], -1, MONTH))
VAR Result = ThisMonth - LastMonth
VAR Arrow =
    IF(
        Result > 0, UNICHAR(9650),
        IF(Result < 0, UNICHAR(9660), UNICHAR(9654))
    ) 
RETURN Arrow & " " & FORMAT(Result, "$##0,K")
